void Tetris::SetColor(int color_num)//设置颜色
{
	int n;
	switch(color_num)
	{
		case 0: n = 0x08; break;
		case 1: n = 0x0C; break;
		case 2: n = 0x0D; break;
		case 3: n = 0x0E; break;
		case 4: n = 0x0A; break;
		case 5: n = 0x0F; break;
		case 6: n = 0x09; break;
		case 7: n = 0x0B; break;
		case 8: n = 0x05; break;
		case 9: n = 0x03; break;
	}
	SetConsoleTextAttribute(GetStdHandle(STD_OUTPUT_HANDLE), n);
}
